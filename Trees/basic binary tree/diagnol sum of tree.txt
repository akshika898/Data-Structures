 static void diagnolSum(Node root)
{
  if(root==null)
    return;
Queue<Node> q=new LinkedList<Node>();
  q.add(root);
while(!q.isEmpty())
{
  c=q.size();
int sum=0;
  while(c>0)
  { 
    Node top=q.remove();
  while(top!=null)
  {
   sum+=top.data;
    if(top.left!=null)
      q.add(top.left);

    top=top.right;
   }
    c--;
  }
System.out.println(sum+" ");
}
