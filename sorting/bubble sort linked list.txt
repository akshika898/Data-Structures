class Solution {
    public ListNode sortList(ListNode head) {
        ListNode i=head,j;
        for(i=head;i!=null;i=i.next)
        {
            for(j=i.next;j!=null;j=j.next)
            {
                if(i.val>j.val){
                    int temp=i.val;
                    i.val=j.val;
                    j.val=temp;
                }
            }
        }
        return head;
    }
}
