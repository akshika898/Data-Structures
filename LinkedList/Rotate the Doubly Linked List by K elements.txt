struct Node* rotateByK(struct Node* head, int k)
{
  if(!head)
  {
    return head;
  }
  struct Node* start1=head;
  int c=0;
  while(start1->next)
  {
    c++;
    start1=start1->next;
  }
  int len=++c;
  start1->next=head;
  head->prev=start1;
  
  start1=head;
  while(k>0)
  {
    start1=start1->prev;
    k--;
  }
  start1->prev->next=NULL;
  start1->prev=NULL;
  head=start1;
  
  return head;
}
