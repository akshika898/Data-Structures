static boolean flag;
static int isFoldable(Node root) {
    if(root==null)
      return 1;
    flag=checkfold(root.left,root.right);
    if(flag==true)
      return 1;
    return 0;
  }
static boolean checkfold(Node t1,Node t2)
  {
    if(t1==null&&t2==null)
      return true;
    else if(t1==null ||t2==null)
      return false;
return checkfold(t1.left,t2.right)&&checkfold(t1.right,t2.left);
  }
