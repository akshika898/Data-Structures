struct Node* newElement(int data)
{
  struct Node* newNode=(struct Node*)malloc(sizeof(struct Node));
  newNode->data=data;
  newNode->next=NULL;
  return newNode;
}

struct Node* insertBeg(struct Node* head, int data)
{
  struct Node* newNode=newElement(data);
  if(head==NULL)
  {
    head=newNode;
    newNode->next=head;
    return head;
  }
  newNode->next=head;
  
  struct Node* temp=head;
  while(temp->next!=head)
  {
    temp=temp->next;
  }
  temp->next=newNode;
  head=newNode;
  
  return head;
}
