Input: [1,12,-5,-6,50,3], k = 4
Output: 12.75
Explanation: Maximum average is 
(12-5-6+50)/4 = 51/4 = 12.75


//o(n)

public class Solution {
    public double findMaxAverage(int[] nums, int k) {
        double sum=0;
        for(int i=0;i<k;i++)
            sum+=nums[i];
        double res=sum;
        for(int i=k;i<nums.length;i++){
            sum+=nums[i]-nums[i-k];
                res=Math.max(res,sum);
        }
        return res/k;
    }
}
