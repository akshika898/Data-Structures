#include <stdio.h>
#include <string.h>

char stack[20];
int top=-1;

int isEmpty()
{
    if(top==-1)
    {
        return 1;
    }
    else
    return 0;
}

void push(char ch)
{
    if(top==20)
    {
        printf("Stack Overflow");
        return;
    }
    top++;
    stack[top]=ch;
}

char pop()
{
    if(isEmpty())
    {
        printf("Stack is empty!");
        return -1;
    }
    return stack[top--];
}

char peek()
{
    return stack[top];
}

int isMatched(char a,char b)
{
    if(a=='(' && b ==')'){
        return 1;
    }
    else if(a=='{' && b =='}'){
        return 1;
    }
    else if(a=='[' && b ==']'){
        return 1;
    }
    else
    return 0;
    
}

int isBalanced(char* str)
{
    char ch;
    for(int i=0;str[i]!='\0';i++){
        ch=str[i];
        if(ch=='{'||ch=='['||ch=='(')
        {
            push(ch);
        }
        else if(ch==')'||ch==']'||ch=='}')
        {
            if(isEmpty())
            {
                return 0;
            }
            else if(isMatched(peek(),ch)==0)
            {
                return 0;
            }
            else{
                pop();
            }
        }
    }
    if(!isEmpty()){
        return 0;
    }
    else
    return 1;
}

int main()
{
    char str[50];
    printf("Enter the string to check :- ");
    scanf("%s",str);
    
    if(isBalanced(str)==0)
    {
        printf("Not Balanced");
    }
    else{
        printf("Balanced");
    }
    
    return 0;
}
