
class Solution {
    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        ListNode a=l1,b=l2;
        ListNode head=new ListNode(0);
        ListNode temp=head;
        while(a!=null &&b!=null){
            if(a.val<b.val)
            {
                temp.next=a;
                a=a.next;
            }
            else
            {
                temp.next=b;
                b=b.next;
            }
            temp=temp.next;
                
        }
        if(a!=null)
            temp.next=a;
        if(b!=null)
            temp.next=b;
        return head.next;
        
    }
}
