class Result {
  static boolean flag;
  static int isFoldable(Node root) {
    if(root==null)
      return 1;
    flag=checkfoldable(root.leftChild,root.rightChild);
    if(flag==true)
      return 1;
    return 0;
  }
  static boolean checkfoldable(Node t1,Node t2)
  {
    if(t1==null&&t2==null)
      return true;
    else if(t1==null ||t2==null)
      return false;
    return checkfoldable(t1.leftChild,t2.rightChild)&&checkfoldable(t1.rightChild,t2.leftChild);
  }
}
