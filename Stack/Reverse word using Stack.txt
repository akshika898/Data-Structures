#include <stdio.h>

char arr[20];
int top=-1;

void push(char ch)
{
    if(top==20)
    {
        printf("Stack Overflow");
        return;
    }
    top++;
    arr[top]=ch;
}

char pop()
{
    if(top==-1)
    {
        printf("Stack is empty!");
        return -1;
    }
    return arr[top--];
}

int peek()
{
    return arr[top];
}

int main()
{
    char str[20];
    printf("Enter the string to reverse :- ");
    scanf("%s",str);
    
    for(int i=0;str[i]!='\0';i++)
    {
        push(str[i]);
    }
    
    for(int i=0;str[i]!='\0';i++)
    {
        printf("%c",pop());
    }
    
    return 0;
}
