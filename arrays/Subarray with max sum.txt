class Solution {
    public int maxSubArray(int[] nums) {

    int max_sum=nums[0],current=0;
    for(int i=0;i<nums.length;i++){ 
    
       current=current+nums[i];
        
        if(max_sum<current){
            max_sum=current;
        }
        if(current<0){
           current=0;
        }
    }
    return max_sum;
 
    }
}
