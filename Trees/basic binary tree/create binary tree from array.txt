static Node buildTree(int a[], int n)
{
  Node root=null;
  root=buildTreeRec(a,root,0,n);
  return root;
}
static Node buildTreeRec(int a[],Node root,int i,int n)
{
  if(i<n)
  {
    root=new Node(a[i]);

    root.left=buildTreeRec(a,root.left,2*i+1,n);
    root.right=buildTreeRec(a,root.right,2*i+2,n);

  }
  return root;
}
