struct Node* deleteBeg(struct Node* head)
{
  if(!head)
  {
    return head;
  }
  struct Node* temp=head;
  while(temp->next!=head)
  {
    temp=temp->next;
  }
  temp->next=head->next;
  head=head->next;
  return head;
}

struct Node* deleteEnd(struct Node* head)
{
  if(!head)
  {
    return head;
  }
  struct Node* temp=head;
  while(temp->next->next!=head)
  {
    temp=temp->next;
  }
  temp->next=head;
  return head;
}
