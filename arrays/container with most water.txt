class Solution {
    public int maxArea(int[] h) {
        int area=0,l=0,r=h.length-1;
        while(l<r)
        {
            area=Math.max(area, Math.min(h[l],h[r])*(r-l));
            if(h[l]<h[r])
                l++;
            else
                r--;
        }
        return area;
    }
}
