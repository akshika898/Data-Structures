class Solution {
    public void moveZeroes(int[] arr) {
        int n=arr.length;
        int temp,c=0;
        for(int i=0;i<n;i++)
        {
            if(arr[i]!=0)
            {
                 temp = arr[c]; 
                 arr[c] = arr[i];    
                 arr[i] = temp; 
                 c = c + 1; 
            }
        }
        
    }
}
