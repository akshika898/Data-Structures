public Node inorderPredecessor(Node root,Node k)
{
     if(root==null||k==null)
        return null;
     Node temp=null;
     while(root!=null)
     {
      if(root.data>=k.data)
       root=root.left;
      else{
        temp=root;
        root=root.right;
          }
     }
    return temp;
}
