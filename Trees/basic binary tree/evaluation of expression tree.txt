static int evaluate(Node t1) {
   if(t1==null)
      return 0;
  if(t1.left==null && t1.right==null)
    return t1.data;
  if(t1.data==43)
    return evaluate(t1.left)+evaluate(t1.right);
  else if(t1.data==45)
    return evaluate(t1.left)-evaluate(t1.right);
  else if(t1.data==42)
    return evaluate(t1.left)*evaluate(t1.right);
  else if(t1.data==47)
    return evaluate(t1.left)/evaluate(t1.right);
 }
