int  loopInList(struct Node* head) 
{
  if(head==NULL){
    return 0;
  }
  int count=1;
  struct Node* slow=head;
  struct Node* fast=head;
  
  while(slow && fast && fast->next)
  {
    slow=slow->next;
    fast=fast->next->next;
    if(slow==fast)
    {
      count=1;
      while(slow->next=fast)
      {
        count++;
        slow=slow->next;
      }
      return count;
    }
  }
  
  return 0;
}
