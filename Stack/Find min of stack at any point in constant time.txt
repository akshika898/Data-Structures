struct stack
{
  int data;
  int min;
};

struct stack s[SIZE];

// Function to add an item to stack.
void push(int item)
{
  if(top==SIZE)
  {
    return;
  }
  if(top!=-1){
    int existMin=s[top].min;
    top++;
    s[top].data=item;
    if(existMin>=item)
    {
      s[top].min=item;
    }
    else
    {
      s[top].data=existMin;
    }
  }
  else{
    top++;
    s[top].data=item;
    s[top].min=item;
  }
}

// Function to remove an item from stack.
int pop()
{
  if(top==-1)
  {
    return -1;
  }
  return s[top--].data;
}

// Function to return the minimum item from stack.
int getMin()
{
  if(top==-1)
  {
    return -1;
  }
  return s[top].min;
}
