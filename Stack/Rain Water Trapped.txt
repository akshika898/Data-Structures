#include <stdio.h>

int min(int a,int b)
{
    if(a>b)
    return b;
    else
    return a;
}

int main()
{
    int n;
    scanf("%d",&n);
    int a[n];
    int res=0;
    
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    
    int l,r;
    for(int i=1;i<n-1;i++)
    {
        r=a[i];
        for(int j=i+1;j<n;j++)
        {
            if(a[j]>r)
            {
                r=a[j];
            }
        }
        
        l=a[i];
        for(int j=i-1;j>=0;j--)
        {
            if(a[j]>l)
            {
                l=a[j];
            }
        }
        
        res = res + min(l,r) - a[i];
    }
    
    printf("%d",res);
    return 0;
}
