int findMiddle(struct Node* head) 
{
  if(head==NULL)
  {
    return -1;
  }
  struct Node* slow=head;
  struct Node* fast=head;
  while(slow && fast && fast->next)
  {
    slow=slow->next;
    fast=fast->next->next;
  }
  return slow->data;
}

int findNLast(struct Node* head, int n)
{
  if(head==NULL)
  {
    return -1;
  }
  struct Node* start=head;
  int c=0;
  while(start->next)
  {
    c++;
    start=start->next;
  }
  c++;
  if(n>c)
  {
    return start->data;
  }
  int x=c-n;
  
  start=head;
  while(x>1)
  {
    x--;
    start=start->next;
  }
  return start->next->data;
}
