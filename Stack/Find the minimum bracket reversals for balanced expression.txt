int minReversal(char* expr)
{
  int l=0,r=0;
  char ch;
  
  for(int i=0;expr[i]!='\0';i++)
  {
    ch=expr[i];
    if(ch=='[')
    {
      l++;
    }
    else if(ch==']')
    {
      if(l==0)
      {
        r++;
      }
      else
      {
        l--;
      }
    }   
  }
  if(l%2!=0 && r==0)
    {
      return -1;
    }
    if(l%2!=0 && r>0)
    {
      return l/2+r+1;
    }
    
    return l/2; 
}
