struct Node* newElement(int data)
{
  struct Node* newNode=(struct Node*)malloc(sizeof(struct Node));
  newNode->data=data;
  newNode->next=NULL;
  return newNode;
}

struct Node* insertSorted(struct Node* head, int data)
{
  struct Node* newNode=newElement(data);
  if(head==NULL)
  {
    head=newNode;
    newNode->next=head;
    return head;
  }
  struct Node* temp=head;
  if(data<head->data)
  {
    newNode->next=head;
    while(temp->next!=head)
    {
      temp=temp->next;
    }
    temp->next=newNode;
    head=newNode;
    return head;
  }
  temp=head->next;
  struct Node* prev=head;
  while(temp!=head)
  {
    if(data<temp->data)
    {
      break;
    }
    prev=temp;
    temp=temp->next;
  }
  newNode->next=prev->next;
  prev->next=newNode;
  
  return head;  
}
