int power(int a,int b)
{
  int res=1;
  for(int i=0;i<b;i++)
  {
    res=res*a;
  }
  return res;
}

int evalPostfix(char* exp)
{
  char ch;
  int a,b;
  for(int i=0;exp[i]!='\0';i++)
  {
    ch=exp[i];
    if(isdigit(ch))
    {
      push(ch - '0');
    }
    else
    {
      b=pop();
      a=pop();
      if(ch=='+')
        push(a+b);
      else if(ch=='-')
        push(a-b);
      else if(ch=='*')
        push(a*b);
      else if(ch=='/')
        push(a/b);
      else if(ch=='^'||ch=='$')
        push(power(a,b));
    }
  }
  return pop();
}
