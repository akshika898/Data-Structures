int isEmpty()
{
  if(main_Stack==NULL)
  {
    return 1;
  }
  else
  {
    return 0;
  }
}

void push(int data)
{
  struct StackLL* new_node=newElement(data);
  new_node->next=main_Stack;
  main_Stack=new_node;
}

int pop()
{
  if(isEmpty())
  {
    return -1;
  }
  int item=main_Stack->data;
  main_Stack=main_Stack->next;
  return item;
}
