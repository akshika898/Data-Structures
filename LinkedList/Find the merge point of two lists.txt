struct Node* findMergePoint(struct Node* head1, struct Node* head2) 
{
  if(!head1||!head2)
  {
    return NULL;
  }
  struct Node* start1=head1;
  struct Node* start2=head2;
  
  while(start1)
  {
    start2=head2;
    while(start2)
    {
      if(start2->data==start1->data)
      {
        return start1;
      }
      start2=start2->next;
    }
    start1=start1->next;
  }
  return NULL;
}
