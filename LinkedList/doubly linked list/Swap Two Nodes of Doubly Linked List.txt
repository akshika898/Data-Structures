void swapNodes(int x, int y)
{
  if(head==null ||x==y)
    return;
  LinkList currx=head,prevx=null,prevy=null,curry=head;
  while(currx!=null && currx.data!=x)
  {
    prevx=currx;
    currx=currx.next;
  }
  while(curry!=null && curry.data!=y)
  {
    prevy=curry;
    curry=curry.next;
  }
  if(currx==null ||curry==null)
    return;
  if(prevx!=null)
    prevx.next=curry;
  else
    head=curry;
  if(prevy!=null)
    prevy.next=currx;
  else
    head=currx;
currx.prev=prevy;
  curry.prev=prevx;
  LinkList temp=curry.next;
  curry.next=currx.next;
  currx.next=temp;
  
}
