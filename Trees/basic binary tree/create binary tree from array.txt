static Node buildTree(int a[], int n)
{
  Node root=null;
  root=buildTreeRec(a,root,0,n);
  return root;
}
static Node buildTreeRec(int a[],Node root,int i,int n)
{
  if(i<n)
  {
    root=new Node(a[i]);
    
    root.leftChild=buildTreeRec(a,root.leftChild,2*i+1,n);
    root.rightChild=buildTreeRec(a,root.rightChild,2*i+2,n);

  }
  return root;
}
