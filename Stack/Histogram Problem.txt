#include <stdio.h>

int main()
{
    int n;
    scanf("%d",&n);
    int a[n];
    int max[n];
    
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    
    int count=0;
    for(int k=0;k<n;k++)
    {
        count=1;
        for(int i=k+1;i<n;i++)
        {
            if(a[i]<a[k])
            break;
            else
            count++;
        }
        
        for(int i=k-1;i>=0;i--)
        {
            if(a[i]<a[k]){
                break;
            }
            else
            count++;
        }
        
        max[k]=count*a[k];
    }
    
    int m=max[0];
    for(int i=0;i<n;i++)
    {
        if(m<max[i]){
            m=max[i];
        }
    }
    printf("%d",m);
    
    return 0;
}
