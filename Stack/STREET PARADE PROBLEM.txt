#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
typedef struct node{
    int  data;
    struct node* next;
}node ;

void push(int val, node** top)
{
    node* nn= (node*)malloc(sizeof(node));
    nn->data=val;
    nn->next=*top;
    *top=nn;
}

void pop(node** top)
{
    struct node* temp=*top;
    *top=temp->next;
    free(temp);
    temp = NULL;

}
int peek(node* top)
{
return top->data;
}

bool empty(node* top)
{
    if(top == NULL)
    return 1;
    return 0;
