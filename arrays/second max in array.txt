import java.util.Scanner;
class Main{
    public static void main(String[] args)
    {
        Scanner sc=new Scanner(System.in);
      int max=Integer.MIN_VALUE,s_max=Integer.MIN_VALUE;
      int a[]=new int[5];
      for(int i=0;i<5;i++)
        a[i]=sc.nextInt();
      for(int i=0;i<5;i++){
        if(a[i]>max){
          s_max=max;
          max=a[i];
        }
        else if(a[i]>s_max &&a[i]!=max){
          s_max=a[i];
        }
      }
      System.out.println(s_max);
    }
}
