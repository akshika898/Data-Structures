LinkList rotateByK(LinkList head, int k)
{
  if(head==null)
    return head;
LinkList temp=head;
  int c=1;
  while(temp.next!=null)
  {
    temp=temp.next;
    c++;
  }
  k=k%c;
  k=c-k;
  temp.next=head;
  temp=head;
  while(k-->1){
    temp=temp.next;
  }
  temp.next.prev=null;
  head=temp.next;
  temp.next=null;
  return head;
}
