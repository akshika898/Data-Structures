static int isMaxHeap(int a[], int n)
{
  int flag=1;
for(int i=n/2-1;i>=0;i--)
  flag=heapify(a,n,i);
return flag;
}
static int heapify(int arr[],int n,int i)
{
  int largest=i;
  int l=2*i+1;
  int r=2*i+2;
  if (l < n && arr[l] > arr[largest] || r < n && arr[r] > arr[largest]) 
  {
    return 0;
  }
 return 1;
}
