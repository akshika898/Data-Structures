class Solution {
    public List<Integer> findDisappearedNumbers(int[] nums) {
        List<Integer> l=new ArrayList<Integer>();
        HashSet<Integer> num=new HashSet<Integer>();
        for(int i:nums)
            num.add(i);
        for(int i=1;i<=nums.length;i++)
        {
            if(!num.contains(i))
                l.add(i);
        }
        return l;
    }
}
