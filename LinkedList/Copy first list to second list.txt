struct Node* newElement(int data)
{
  struct Node* newNode=(struct Node*)malloc(sizeof(struct Node));
  newNode->data=data;
  newNode->next=NULL;
  return newNode;
}

struct Node *copyList(struct Node *org)
{
  struct Node* head=NULL;
  while(org)
  {
    struct Node* newNode=newElement(org->data);
    if(head==NULL)
    {
      head=newNode;
    }
    else
    {
      struct Node* ptr=head;
      while(ptr->next)
      {
        ptr=ptr->next;
      }
      ptr->next=newNode;
    }
    org=org->next;
  }
  return head;
}
